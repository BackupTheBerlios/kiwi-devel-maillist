This patch is to add priority to the repository to the boot config of RHEL iso images. 
This will resolve the conflicts with the kiwi-tools package while creating the boot image. 




diff -up system/boot/ix86/isoboot/rhel-05.4/config.xml.old system/boot/ix86/isoboot/rhel-05.4/config.xml
--- system/boot/ix86/isoboot/rhel-05.4/config.xml.old	2011-09-02 15:41:36.739059696 -0500
+++ system/boot/ix86/isoboot/rhel-05.4/config.xml	2011-09-02 15:41:57.794796464 -0500
@@ -61,7 +61,7 @@
 	<repository type="rpm-md" status="replaceable">
 		<source path="iso:///image/CDs/RHEL-5.4-%arch-DVD.iso/Server"/>
 	</repository>
-	<repository type="rpm-md" status="fixed">
+	<repository type="rpm-md" status="fixed" priority="1">
 		<source path="/usr/share/kiwi/repo/rhel-repo/rhel-054-repo"/>
 	</repository>
 	<packages type="bootstrap" profiles="std">
